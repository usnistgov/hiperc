27 void device_boundaries(struct OpenCLData* dev, const int flip,
28  const int nx, const int ny, const int nm,
29  const int bx, const int by)
30 {
31  /* OpenCL uses cl_mem, not fp_t*, so swap_pointers won't work.
32  We leave the pointers alone but call the kernel on the appropriate data location.
33  */
34  const cl_mem d_conc_old = (flip == 0) ? dev->conc_old : dev->conc_new;
35 
36  const size_t tile_dim[2] = {bx, by};
37  const size_t bloc_dim[2] = {ceil((float)(nx) / (tile_dim[0] - nm + 1)),
38  ceil((float)(ny) / (tile_dim[1] - nm + 1))
39  };
40  const size_t grid_dim[2] = {bloc_dim[0]* tile_dim[0],
41  bloc_dim[1]* tile_dim[1]
42  };
43 
44  cl_int status = clSetKernelArg(dev->boundary_kernel, 0, sizeof(cl_mem), (void*)&d_conc_old);
45  status |= clSetKernelArg(dev->boundary_kernel, 1, sizeof(int), (void*)&nx);
46  status |= clSetKernelArg(dev->boundary_kernel, 2, sizeof(int), (void*)&ny);
47  status |= clSetKernelArg(dev->boundary_kernel, 3, sizeof(int), (void*)&nm);
48  report_error(status, "setting boundary kernel args");
49 
50  status |= clEnqueueNDRangeKernel(dev->commandQueue, dev->boundary_kernel, 2, NULL, grid_dim, tile_dim, 0, NULL, NULL);
51  report_error(status, "enqueuing boundary kernel");
52 }
