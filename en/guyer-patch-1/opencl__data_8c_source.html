52  /* Platform: The host plus a collection of devices managed by a specific
53  OpenCL implementation that allow an application to share resources and
54  execute kernels on devices in the platform. There will typically be one
55  platform, e.g. nvidia-opencl or amd-opencl. It is not uncommon to have
56  two or three cards from different manufacturers installed, with one
57  platform per vendor, so "typical" is not "only."
58  */
59  status = clGetPlatformIDs(0, NULL, &numPlatforms);
60  report_error(status, "clGetPlatformIDs");
61  if (numPlatforms == 0) {
62  printf("Error: No OpenCL framework found.\n");
63  exit(-1);
64  }
65  platforms = (cl_platform_id*)malloc(numPlatforms);
66  status = clGetPlatformIDs(numPlatforms, platforms, NULL);
67  report_error(status, "clGetPlatformIDs");
68 
69  /* Context: shared memory space, allowing threads to collaborate.
70  Specific to one platform.
71  */
72  cl_context_properties properties[] = {CL_CONTEXT_PLATFORM,
73  (cl_context_properties)platforms[0],
74  0
75  };
76  dev->context = clCreateContextFromType(properties, CL_DEVICE_TYPE_ALL, 0, NULL, &status);
77  report_error(status, "clCreateContextFromType");
78 
79  /* Device: CPU, GPU, FPGA, or other threaded hardware available to do work.
80  Specific to one context. If you have more than one device, the following
81  definition of "gpu" may fail. Contact the developers if you'd like to
82  help us generalize the code to properly handle your setup.
83  */
84  status = clGetContextInfo(dev->context, CL_CONTEXT_DEVICES, 0, NULL, &numDevices);
85  report_error(status, "clGetContextInfo");
86  devices = (cl_device_id*)malloc(numDevices);
87  status = clGetContextInfo(dev->context, CL_CONTEXT_DEVICES, numDevices, devices, NULL);
88  report_error(status, "clGetContextInfo");
89  if (numDevices == 0) {
90  printf("Error: No OpenCL-compatible devices found.\n");
91  exit(-1);
92  }
93  gpu = devices[0];
94 
95  /* CommandQueue: coordinator of kernels to run in a given context */
96  /* OpenCL v1 */
97  dev->commandQueue = clCreateCommandQueue(dev->context, gpu, 0, &status);
98  report_error(status, "clCreateCommandQueue");
99  /* OpenCL v2 */
100  /* dev->commandQueue = clCreateCommandQueueWithProperties(dev->context, gpu, (cl_queue_properties*)NULL, &status); */
101  /* report_error(status, "clCreateCommandQueue"); */
102 
103  /* Program: set of one or more kernels to run in a given context,
104  read from kernel files *.cl
105  */
106  build_program("kernel_boundary.cl", &(dev->context), &gpu, &(dev->boundary_program), &status);
107  build_program("kernel_convolution.cl", &(dev->context), &gpu, &(dev->convolution_program), &status);
108  build_program("kernel_diffusion.cl", &(dev->context), &gpu, &(dev->diffusion_program), &status);
109 
110  /* Kernel: code compatible with the just-in-time (JIT) compiler
111  Names (strings) must match the name of a function defined in the
112  cl_program produced by build_program().
113  */
114  dev->boundary_kernel = clCreateKernel(dev->boundary_program, "boundary_kernel", &status);
115  report_error(status, "dev->boundary_kernel");
116  dev->convolution_kernel = clCreateKernel(dev->convolution_program, "convolution_kernel", &status);
117  report_error(status, "dev->convolution_kernel");
118  dev->diffusion_kernel = clCreateKernel(dev->diffusion_program, "diffusion_kernel", &status);
119  report_error(status, "dev->diffusion_kernel");
120 
121  /* allocate memory on device */
122  dev->conc_old = clCreateBuffer(dev->context, CL_MEM_READ_WRITE | CL_MEM_COPY_HOST_PTR, gridSize, conc_old[0], &status);
123  report_error(status, "create dev->conc_old");
124  dev->conc_new = clCreateBuffer(dev->context, CL_MEM_READ_WRITE | CL_MEM_ALLOC_HOST_PTR, gridSize, NULL, &status);
125  report_error(status, "create dev->conc_new");
126  dev->conc_lap = clCreateBuffer(dev->context, CL_MEM_READ_WRITE, gridSize, NULL, &status);
127  report_error(status, "create dev->conc_lap");
128 
129  dev->mask = clCreateBuffer(dev->context, CL_MEM_READ_ONLY | CL_MEM_COPY_HOST_PTR, maskSize, mask_lap[0], &status);
130  report_error(status, "create dev->mask");
131 
132  /* clean up */
133  free(platforms);
134  free(devices);
135 }

137 void build_program(const char* filename,
138  cl_context* context,
139  cl_device_id* gpu,
140  cl_program* program,
141  cl_int* status)
142 {
143  FILE* fp;
144  char* source_str;
145  char msg[1024];
146  size_t source_len, program_size, read_size;
147  char options[] = "-I../common-diffusion";
148 
149  fp = fopen(filename, "rb");
150  if (!fp) {
151  printf("Failed to load kernel %s\n", filename);
152  exit(-1);
153  }
154 
155  fseek(fp, 0, SEEK_END);
156  program_size = ftell(fp);
157  rewind(fp);
158 
159  source_str = (char*)malloc(program_size + sizeof(char));
160  source_str[program_size] = '\0';
161 
162  read_size = fread(source_str, sizeof(char), program_size, fp);
163  assert(read_size == program_size);
164  fclose(fp);
165 
166  strcpy(msg, filename);
167  strcat(msg, ": clCreateProgramWithSource");
168  source_len = strlen(source_str);
169  *program = clCreateProgramWithSource(*context, 1, (const char**)&source_str, &source_len, status);
170  report_error(*status, msg);
171 
172  strcpy(msg, filename);
173  strcat(msg, ": clBuildProgram");
174  *status = clBuildProgram(*program, 0, NULL, (const char*)options, NULL, NULL);
175 
176  /* report_error is too granular: report specific build errors */
177  if (*status != CL_SUCCESS) {
178  /* Thanks to https://stackoverflow.com/a/29813956 */
179  char* buff_erro;
180  cl_int errcode;
181  size_t build_log_len;
182  errcode = clGetProgramBuildInfo(*program, *gpu, CL_PROGRAM_BUILD_LOG, 0, NULL, &build_log_len);
183  if (errcode) {
184  printf("clGetProgramBuildInfo failed at line %d\n", __LINE__);
185  exit(-1);
186  }
187 
188  buff_erro = (char*)malloc(build_log_len);
189  if (!buff_erro) {
190  printf("malloc failed at line %d\n", __LINE__);
191  exit(-2);
192  }
193 
194  errcode = clGetProgramBuildInfo(*program, *gpu, CL_PROGRAM_BUILD_LOG, build_log_len, buff_erro, NULL);
195  if (errcode) {
196  printf("clGetProgramBuildInfo failed at line %d\n", __LINE__);
197  exit(-3);
198  }
199 
200  fprintf(stderr,"Build log: \n%s\n", buff_erro);
201  free(buff_erro);
202  fprintf(stderr,"clBuildProgram failed\n");
203  }
204 
205  report_error(*status, msg);
206 
207  /* clean up */
208  free(source_str);
209 }
