96 fp_t distance_point_to_segment(const fp_t ax, const fp_t ay,
97  const fp_t bx, const fp_t by,
98  const fp_t px, const fp_t py)
99 {
100  const fp_t L2 = (ax - bx) * (ax - bx) + (ay - by) * (ay - by);
101  if (L2 == 0.) /* line segment is just a point */
102  return euclidean_distance(ax, ay, px, py);
103  const fp_t t = fmax(0., fmin(1., ((px - ax) * (bx - ax) + (py - ay) * (by - ay)) / L2));
104  const fp_t zx = ax + t * (bx - ax);
105  const fp_t zy = ay + t * (by - ay);
106  return euclidean_distance(px, py, zx, zy);
107 }

114 void check_solution(fp_t** conc_new, fp_t** conc_lap, const int nx, const int ny, const fp_t dx, const fp_t dy, const int nm,
115  const fp_t elapsed, const fp_t D, fp_t* rss)
116 {
117  fp_t sum = 0.;
118  int i, j;
119 
120  #ifdef __OPENMP
121  #pragma omp parallel reduction(+:sum)
122  {
123  #pragma omp for collapse(2) private (i,j)
124  #endif
125  for (j = nm/2; j < ny-nm/2; j++) {
126  for (i = nm/2; i < nx-nm/2; i++) {
127  fp_t cal, car, r;
128 
129  /* numerical solution */
130  const fp_t cn = conc_new[j][i];
131 
132  /* shortest distance to left-wall source */
133  r = distance_point_to_segment(dx * (nm/2), dy * (nm/2),
134  dx * (nm/2), dy * (ny/2),
135  dx * i, dy * j);
136  analytical_value(r, elapsed, D, &cal);
137 
138  /* shortest distance to right-wall source */
139  r = distance_point_to_segment(dx * (nx-1-nm/2), dy * (ny/2),
140  dx * (nx-1-nm/2), dy * (ny-1-nm/2),
141  dx * i, dy * j);
142  analytical_value(r, elapsed, D, &car);
143 
144  /* superposition of analytical solutions */
145  const fp_t ca = cal + car;
146 
147  /* residual sum of squares (RSS) */
148  conc_lap[j][i] = (ca - cn) * (ca - cn) / (fp_t)((nx-1-nm/2) * (ny-1-nm/2));
149  }
150  }
151 
152  #ifdef __OPENMP
153  #pragma omp for collapse(2) private(i,j)
154  #endif
155  for (j = nm/2; j < ny-nm/2; j++) {
156  for (i = nm/2; i < nx-nm/2; i++) {
157  sum += conc_lap[j][i];
158  }
159  }
160  #ifdef __OPENMP
161  }
162  #endif
163 
164  *rss = sum;
165 }
