23  const int nx, const int ny, const int nm)
24 {
25  /* Lambda function executed on each thread, solving convolution */
26  tbb::parallel_for(tbb::blocked_range2d<int>(nm/2, nx-nm/2, nm/2, ny-nm/2),
27  [=](const tbb::blocked_range2d<int>& r) {
28  for (int j = r.cols().begin(); j != r.cols().end(); j++) {
29  for (int i = r.rows().begin(); i != r.rows().end(); i++) {
30  fp_t value = 0.0;
31  for (int mj = -nm/2; mj < nm/2+1; mj++) {
32  for (int mi = -nm/2; mi < nm/2+1; mi++) {
33  value += mask_lap[mj+nm/2][mi+nm/2] * conc_old[j+mj][i+mi];

90

102
