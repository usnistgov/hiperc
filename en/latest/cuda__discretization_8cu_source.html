28 __global__ void convolution_kernel(fp_t* d_conc_old, fp_t* d_conc_lap,
29  const int nx, const int ny, const int nm)
30 {
31  int dst_x, dst_y, dst_nx, dst_ny;
32  int src_x, src_y, src_nx, src_ny;
33  int til_x, til_y, til_nx;
34  fp_t value=0.;
35 
36  /* source and tile width include the halo cells */
37  src_nx = blockDim.x;
38  src_ny = blockDim.y;
39  til_nx = src_nx;
40 
41  /* destination width excludes the halo cells */
42  dst_nx = src_nx - nm + 1;
43  dst_ny = src_ny - nm + 1;
44 
45  /* determine tile indices on which to operate */
46  til_x = threadIdx.x;
47  til_y = threadIdx.y;
48 
49  dst_x = blockIdx.x * dst_nx + til_x;
50  dst_y = blockIdx.y * dst_ny + til_y;
51 
52  src_x = dst_x - nm/2;
53  src_y = dst_y - nm/2;
54 
55  /* copy tile: __shared__ gives access to all threads working on this tile */
56  extern __shared__ fp_t d_conc_tile[];
57 
58  if (src_x >= 0 && src_x < nx &&
59  src_y >= 0 && src_y < ny ) {
60  /* if src_y==0, then dst_y==nm/2: this is a halo row */
61  d_conc_tile[til_nx * til_y + til_x] = d_conc_old[nx * src_y + src_x];
62  }
63 
64  /* tile data is shared: wait for all threads to finish copying */
65  __syncthreads();
66 
67  /* compute the convolution */
68  if (til_x < dst_nx && til_y < dst_ny) {
69  for (int j = 0; j < nm; j++) {
70  for (int i = 0; i < nm; i++) {
71  value += d_mask[j * nm + i] * d_conc_tile[til_nx * (til_y+j) + til_x+i];
72  }
73  }
74  /* record value */
75  if (dst_y < ny && dst_x < nx) {
76  d_conc_lap[nx * dst_y + dst_x] = value;
77  }
78  }
79 
80  /* wait for all threads to finish writing */
81  __syncthreads();
82 }

85 __global__ void diffusion_kernel(fp_t* d_conc_old, fp_t* d_conc_new, fp_t* d_conc_lap,
86  const int nx, const int ny, const int nm,
87  const fp_t D, const fp_t dt)
88 {
89  int thr_x, thr_y, x, y;
90 
91  /* determine indices on which to operate */
92  thr_x = threadIdx.x;
93  thr_y = threadIdx.y;
94 
95  x = blockDim.x * blockIdx.x + thr_x;
96  y = blockDim.y * blockIdx.y + thr_y;
97 
98  /* explicit Euler solution to the equation of motion */
99  if (x < nx && y < ny) {
100  d_conc_new[nx * y + x] = d_conc_old[nx * y + x]
101  + dt * D * d_conc_lap[nx * y + x];
102  }
103 
104  /* wait for all threads to finish writing */
105  __syncthreads();
106 }
